import { type ClassValue, clsx } from "clsx";
import { twMerge } from "tailwind-merge";

export function cn(...inputs: ClassValue[]) {
  return twMerge(clsx(inputs));
}

// Format currency for GCC countries
export function formatCurrency(
  amount: number,
  country: "KW" | "SA" | "AE" | "QA" | "BH" | "OM" = "KW"
) {
  const config: Record<string, { code: string; decimals: number; locale: string }> = {
    KW: { code: "KWD", decimals: 3, locale: "ar-KW" },
    SA: { code: "SAR", decimals: 2, locale: "ar-SA" },
    AE: { code: "AED", decimals: 2, locale: "ar-AE" },
    QA: { code: "QAR", decimals: 2, locale: "ar-QA" },
    BH: { code: "BHD", decimals: 3, locale: "ar-BH" },
    OM: { code: "OMR", decimals: 3, locale: "ar-OM" },
  };

  const { code, decimals, locale } = config[country];

  return new Intl.NumberFormat(locale, {
    style: "currency",
    currency: code,
    minimumFractionDigits: decimals,
  }).format(amount);
}

// Format date in Arabic
export function formatDate(date: Date, style: "full" | "long" | "medium" | "short" = "long") {
  return new Intl.DateTimeFormat("ar", {
    dateStyle: style,
  }).format(date);
}

// Format time in Arabic
export function formatTime(date: Date) {
  return new Intl.DateTimeFormat("ar", {
    hour: "numeric",
    minute: "numeric",
  }).format(date);
}

// Validate GCC phone number
export function validatePhone(phone: string, country: string): boolean {
  const patterns: Record<string, RegExp> = {
    KW: /^[569]\d{7}$/,
    SA: /^5\d{8}$/,
    AE: /^5\d{8}$/,
    QA: /^[3567]\d{7}$/,
    BH: /^[3]\d{7}$/,
    OM: /^[79]\d{7}$/,
  };

  const digits = phone.replace(/\D/g, "");
  return patterns[country]?.test(digits) ?? false;
}

// Calculate VAT
export function calculateVAT(amount: number, country: string): number {
  const rates: Record<string, number> = {
    KW: 0,
    SA: 0.15,
    AE: 0.05,
    QA: 0,
    BH: 0.10,
    OM: 0.05,
  };

  return amount * (rates[country] ?? 0);
}
